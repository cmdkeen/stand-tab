program worlds {
teams as System.getTeams()
visible function createRooms (goals scores) {
pools as Pool(teams, speakerPoints) 
while(pools.notEmpty()) {
	t as pools.pop() 
	while(t.size()%4!=0) {
		v as pools.pop()
		if(v.size()<(t.size()%4)) {
			a = Pullup(v)
			t.add(a)
			pools.push(v)
		} else {
			t.add(v)
		}
	}
	Allocate(t,rooms)
}
}
}
